canStart.ts 
function canStart(op) {
  const blockers = [];
  
  if (!op.rtwVerified)
    blockers.push('RTW');
  if (op.cscsExpiry < now)
    blockers.push('CSCS');
  if (op.status === 'blocked')
    blockers.push('Blocked');
  
  return blockers.length === 0;
}
